=================================================================

int main() {
    int arr[7] = {10, 20, 30, 40, 25, 50, 45};
    int i = 4;
    int j = i - 1;
    int key = arr[i];

    while (j >= 0 && arr[j] > key) {
        arr[j + 1] = arr[j];
        j--;
    }
    arr[j + 1] = key;
    cout << arr[1] << endl;
    cout << arr[2] << endl;
    cout << arr[3] << endl;
    cout << arr[4] << endl;    

    return 0;
}
